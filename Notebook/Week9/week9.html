sales = [[100,410,120],[430,120,20],[190,184,420],[330,90,280]]

persons = ['Sato','Ueda','Mori','Suzuki']

w = 0.5  
x = np.arange(len(types)) 

btm = np.zeros(len(types))

for z in range(len(persons)):
    plt.bar(x, sales[z], width=w, label=persons[z], bottom=btm)
    btm += np.array(sales[z])

plt.ylim(0,1500) 
plt.xticks(x,types)
plt.title('Sales Results')
plt.xlabel('Type of Products')
plt.ylabel('Amount of Sales')
plt.legend(loc = "upper right")
