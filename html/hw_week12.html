def calc_corr(x, y):
    if len(x) != len(y):
        print('length error')
        return 0
    
    xm = calc_mean(x)
    ym = calc_mean(y)
    
    #  (yx = ...)
    yx = [ (x[i]-xm) * (y[i]-ym) for i in range(len(x)) ]
    
    xs = math.sqrt(calc_var(x))
    ys = math.sqrt(calc_var(y))
    
    # 0
    if xs * ys == 0:
        return 0
        
    return calc_mean(yx) / (xs * ys)
