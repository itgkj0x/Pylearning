X = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec']
X0 = np.arange(len(X)) 
Y1 = [9.6,10.4,13.6,19.0,22.9,25.5,29.2,30.8,26.9,21.5,16.3,11.9]
Y2 = [5.2,5.7,8.7,13.9,18.2,21.4,25.0,26.4,22.8,17.5,12.1,7.6]
Y3 = [0.9,1.7,4.4,9.4,14.0,18.0,21.8,23.0,19.7,14.2,8.3,3.5]

w = 0.4

plt.bar(X0-0.2,Y1,w,label='High')  
plt.bar(X0,Y2,w,label='Mean')
plt.bar(X0+0.2,Y3,w,label='Low')   

plt.xticks(X0,X)   
plt.title('Temperatures in Tokyo(1981-2010)')
plt.xlabel('Month')
plt.ylabel('Celsius degree')
plt.legend()
plt.show()

演習3¶

ある商店の販売員4名の今月の売り上げ結果が以下の表に示されている。売り上げの販売員ごとの内訳も分かるような以下の販売実績グラフを表示するプログラムを作成せよ。ただし以下の変数を使うこと。

sales = [[100,410,120],[430,120,20],[190,184,420],[330,90,280]]
types = ['Office','Home','Hobby']
persons = ['Sato','Ueda','Mori','Suzuki']
In [34]:
sales = [[100,410,120],[430,120,20],[190,184,420],[330,90,280]]
types = ['Office','Home','Hobby']
persons = ['Sato','Ueda','Mori','Suzuki']

w = 0.5  
x = np.arange(len(types)) 

btm = np.zeros(len(types))

for z in range(len(persons)):
    plt.bar(x, sales[z], width=w, label=persons[z], bottom=btm)
    btm += np.array(sales[z])

plt.ylim(0,1500) 
plt.xticks(x,types)
plt.title('Sales Results')
plt.xlabel('Type of Products')
plt.ylabel('Amount of Sales')
plt.legend(loc = "upper right")
plt.show()
